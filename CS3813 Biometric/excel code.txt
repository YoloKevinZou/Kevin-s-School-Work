Function checkEqual(a, b, score, threshold)
    
    If a = b Then
        If score > threshold Then
            checkEqual = 1
        Else
            checkEqual = 1000
        End If
       
    Else
        checkEqual = 0
    End If
    
End Function

Function CheckNotEqual(a, b, score, threshold)

    If a <> b Then
        
        If score > threshold Then
            CheckNotEqual = 1000
        Else
            CheckNotEqual = 1
        End If
    Else
        CheckNotEqual = 0
    
    
    End If

End Function
